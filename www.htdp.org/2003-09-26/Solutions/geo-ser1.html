Problem Statement 
#| ---------------------------------------------------------------------------
   g-fives : N -> number
   to compute the n-the term in the geometric series
     3, 15, 75, 375, ...
   recursive computation

   Examples:
    (g-fives 0) = 3
    (g-fives 2) = 75
|#

(define (g-fives n)
  (cond
    ((zero? n) 3)
    (else (* 5 (g-fives (sub1 n))))))
