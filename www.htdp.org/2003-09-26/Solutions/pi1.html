;; ! : N -> N
;; to compute the product of the numbers from n to 1
(define (! n) 
  (cond 
    [(zero? n) 1] 
    [else (* n (! (sub1 n)))]))

;; product : N N -> N
;; to compute the product of the numbers from n (exclusive) m (inclusive)
(define (product n m)
  (/ (! m)
     (! n)))

;; EXAMPLES AS TESTS
(product 11 12) "should be" 12
(product 2 5) "should be" 60
