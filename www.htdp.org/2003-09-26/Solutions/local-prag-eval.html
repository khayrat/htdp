;; mysort : list-of-numbers  ->  list-of-numbers
(define (mysort alon)
  (local ((define (mysort alon)
            (cond
              [(empty? alon) empty]
              [(cons? alon) (insert (first alon)
                                    (mysort (rest alon)))]))
          (define (insert an alon)
            (cond
              [(empty? alon) (list an)]
              [else (cond
                      [(> an (first alon)) (cons an alon)]
                      [else (cons (first alon) 
                                  (insert an (rest alon)))])])))
    (mysort alon)))
(mysort (list 2 1 3))
