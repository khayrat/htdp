          (cond 
            [(zero? an) false] 
            [else (even (sub1 an))])) 
        (define (even an) 
          (cond 
            [(zero? an) true] 
            [else (odd (sub1 an))]))) 
  (even a-nat-num)) 

;;   RED: odd and even are the locally defined variables
;; GREEN: these are the right-hand sides:
   (cond 
     [(zero? an) false] 
     [else (even (sub1 an))])

   (cond 
     [(zero? an) true] 
     [else (odd (sub1 an))])
