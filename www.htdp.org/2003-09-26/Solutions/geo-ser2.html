Problem Statement 
#| ---------------------------------------------------------------------------
   g-fives : N -> number
   to compute the n-the term in the geometric series
     3, 15, 75, 375, ...
   closed form

   Examples:
    (g-fives-closed 0) = 3
    (g-fives-closed 2) = 75
|#

(define (g-fives-closed n)
  (* 3 (expt 5 n)))
