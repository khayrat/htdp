;; Language: Intermediate Student with Lambda

;; [List-of Number] [List-of Number] -> [List-of Number]
;; subtract the first list from the second so that the first number is 0
;; ASSUME: the two lists are of equal length and there is at least one number

(check-expect (subtract '(1 1 1) '(3 8 10)) `(,(- 8 3) ,(- 10 3)))
(check-expect (subtract '(1 2 3) '(3 8 10)) `(,(- 8 (* 3 2)) ,(- 10 (* 3 3))))

(define (subtract u v)
  (local ((define u1 (first u))
          (define v1 (first v))
          (define r (/ v1 u1)))
    (rest (map - v (map (lambda (x) (* x r)) u)))))
