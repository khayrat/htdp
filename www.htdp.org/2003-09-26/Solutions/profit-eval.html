;; How to design a program  
(define (profit-left ticket-price) 
  (- (revenue ticket-price) 
     (cost ticket-price))) 

(define (revenue ticket-price) 
  (*  (attendees ticket-price) ticket-price)) 

(define (cost ticket-price) 
  (+ 180  
     (* .04 (attendees ticket-price)))) 

(define (attendees ticket-price) 
  (+ 120 
     (* (/ 15 .10) (- 5.00 ticket-price)))) 


;; How not to design a program  
(define (profit-right price) 
  (- (* (+ 120 
           (* (/ 15 .10) 
              (- 5.00 price))) 
        price) 
     (+ 180  
        (* .04 
           (+ 120 
              (* (/ 15 .10) 
                 (- 5.00 price))))))) 


;; results for original profit-left and profit-right
(profit-left 3.00) "should be"
(profit-right 3.00)
(profit-left 4.00) "should be"
(profit-right 4.00)
(profit-left 5.00) "should be"
(profit-right 5.00)
